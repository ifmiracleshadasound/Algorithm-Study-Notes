## easy
```C++

typedef long long LL;
const int N = 2001, MOD = 1e9+7;
LL f[N][N];

//杨辉三角
void cal(){
    f[0][0] = 1;
    for( int i=1; i<N; i++ ){
        f[i][0] = 1;
        for( int j=1; j<=i; j++ ){
            f[i][j] = (f[i-1][j] + f[i-1][j-1]) % MOD;
        }
    }
}
```

```C++
typedef long long LL;
const int N = 100001, MOD = 1e9+7;

LL f[N], rvf[N];  //f[a] = a!(mod MOD)  rvf[b] = (b!)^(−1) (mod MOD)
LL quickPow(LL a, LL b, LL p){
    LL res = 1 % p;
    while( b ){
        if( b & 1 ) res = res * a % p;
        a = a * a % p;
        b >>= 1;
    }
    return res;
}
void cal(){
    f[0] = rvf[0] = 1;
    for( int i=1; i<N; i++ ){
        f[i] = f[i-1] * i % MOD;
        rvf[i] = rvf[i-1] * quickPow(i, MOD-2, MOD) % MOD;
    }
}
```

```C++
typedef long long LL;
LL quickPow(LL a, LL b, LL p){
    LL res = 1 % p;
    while( b ){
        if( b & 1 ) res = res * a % p;
        a = a * a % p;
        b >>= 1;
    }
    return res;
}
int C(LL a, LL b, LL p){
    LL res = 1;
    for( int i=1, j=a; i<=b; i++, j-- ){
        res = res * j % p * quickPow(i, p-2, p) % p;
    }
    return res;
}
int lucas(LL a, LL b, LL p){
    if( a < p && b < p ) return C(a, b, p);
    return (LL)C(a%p, b%p, p) * lucas(a/p, b/p, p) % p;
}
```
