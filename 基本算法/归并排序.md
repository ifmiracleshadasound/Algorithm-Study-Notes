## 归并排序
```c++
const int N = 100010;
int temp[N];

void MergeSort(int a[], int l, int r){
    if( l>=r ) return;
    int mid = ( l + r ) >> 1;
    MergeSort(a, l, mid);
    MergeSort(a, mid+1, r);
    int i=l, j=mid+1, k=0;
    while( i<=mid && j<=r ){
        if( a[i]<=a[j] ) temp[k++] = a[i++];
        else temp[k++] = a[j++];
    }
    while( i<=mid ) temp[k++] = a[i++];
    while( j<=r ) temp[k++] = a[j++];
    for( i=l; i<=r; i++ ) a[i] = temp[i-l];
    
}
//usage: MergeSort(a, 0, n-1);
```
