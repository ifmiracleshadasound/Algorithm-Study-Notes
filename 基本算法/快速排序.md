const int N = 100010;
int a[N], n;
//快速排序 -- 分治思想 
void quickSort(int l, int r){
    if( l>=r ) return; //边界返回
    int i = l-1, j = r+1, x = a[l+r>>1]; //1.确定分界点 随机取出一个数
    //用x保存a[l+r>>1]而不用下标mid=l+r>>1，mid中的值可能会在交换的过程中被改变
    while( i<j ){
        //2.调整区间
        while( a[++i]<x );
        while( a[++j]>x );
        if( i<j ) swap(a[i], a[j]);
    }
    //3.递归处理左右两端
    quickSort(l, j);
    quickSort(j+1, r);
}
