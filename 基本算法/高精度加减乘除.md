## 加
```c++
vector<int> add(const vector<int>& A, const vector<int>& B){
    vector<int> res;
    res.reserve(A.size()+B.size());
    for( int i=0, c=0; i<A.size() || i<B.size() || c; i++ ){
        if( i<A.size() ) c += A[i];
        if( i<B.size() ) c += B[i];
        res.push_back(c%10);
        c /= 10;
    }
    return res;
}
```

## 减
```c++
/***************************************************
//judge A and B before sub(A, B)
//A>B -> sub(A, B)
//A<B -> sub(B, A)
****************************************************/
int cmp(const vector<int>& A, const vector<int>& B){
    if( A.size() > B.size() ) return 1;
    if( A.size() < B.size() ) return -1;
    for( int i=A.size()-1; i>=0; i-- ){
        if( A[i]>B[i] ) return 1;
        else if( A[i]<B[i] ) return -1;
    }
    return 0;
}


vector<int> sub(const vector<int>& A, const vector<int>& B){
    vector<int> res;
    res.reserve(A.size());
    for( int i=0, c=0; i<A.size(); i++ ){
        c += A[i];
        if( i<B.size() ) c -= B[i];
        res.push_back((c+10)%10);
        if( c<0 ) c = -1;
        else c= 0;
    }
    while( res.size()>1 && res.back()==0 ) res.pop_back();
    return res;
}
```

## 乘

```c++
vector<int> multi(const vector<int>& A, const vector<int>& B){
    vector<int> res(A.size()+B.size());
    for( int i=0; i<A.size(); i++ ){
        for( int j=0; j<B.size(); j++ ){
            res[i+j] += A[i] * B[j];
        }
    }
    for( int i=0, c=0; i<res.size(); i++ ){
        res[i] += c;
        c = res[i] / 10;
        res[i] %= 10;
    }
    while( res.size()>1 && res.back()==0 ) res.pop_back();
    return res;
}
```
